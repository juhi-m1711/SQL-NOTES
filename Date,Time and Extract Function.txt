-- 1. Find all orders placed in February

from orders

-- 2. Find all orders placed in 2024

from orders

-- 3. Calculate number of days between order and delivery
select *,
datediff(delivery_date, order_date) as days_to_deliver
from orders;

-- 4. Find orders placed after 6 PM
select *
from orders
where hour(order_time) > 18 ;

-- 5. Combine date and time into a DATETIME
select *,
concat(order_date, ' ', order_time) as order_datetime
from orders;

-- 6. Sort by time (earliest to latest)
select * 
from orders
order by order_time asc;

-- 7. Display orders where delivery took more than 5 days.
select *,
datediff(delivery_date, order_date) as days_to_deliver
from orders
where datediff(delivery_date, order_date) > 5;

-- 8. Extract the day name (Monday, Tuesday, etc.) from the order date.

from orders;

monthname(order_date) as order_monthname
from orders;

-- 11. Subtract 3 days from delivery date.
