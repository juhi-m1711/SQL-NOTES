create database triggers;
use triggers;

create table employees (
	emp_id int primary key,
    emp_name varchar(100),
    emp_age int,
    gender varchar(50),
    emp_salary int 
    );

create trigger before_salary_update
before insert 
on employees
for each row
	set new.emp_salary = new.emp_salary + 5000 ;
