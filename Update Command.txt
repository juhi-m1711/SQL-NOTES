select * from employees_info;
describe employees_info;

--Updating a column of a single row
update employees info set
	age = 30 where id = 3;

-- Updating Different specific columns of a single row
update employees_info set 
	name = 'Juhi', age = 20 where id = 4;

-- Updating Multiple Rows dynamically
update employees_info 
set experience = case
	when experience = 4 then 5
    	when experience = 1 then 5
    	when experience = 3 then 5
    	when experience = 10 then 5
	else experience
end 
where id in (3,4,6,9) and (experience in (4,1,3,10));

-- Updating multiple columns of multiple rows
update employees_info
set experience= case
	when experience = 4 then 5
    	when experience = 1 then 5
    	when experience = 3 then 5
    	when experience = 10 then 5
	else experience
end,
	salary= case
		when salary=75000 then 90000
		else salary
end
where id in (3,4,6,9) and (experience in (4,1,3,10) or salary = 75000);
